\subsection{Weymouth-Yue}
Original volume plus the boundary flux (Eulerian).
\begin{equation}
  \label{eq:wy}
  \tilde{f_{c}} = f_{c} + VOF^{1}_{c} - VOF^{3}_{c} - VOF^{1}_{r} + VOF^{3}_{l}
\end{equation}
